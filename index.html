Free Pascal Tutorial: Procedural Programming Keywords

program

Description: The program keyword defines the name of the program and is used to begin the main block of the code.

program HelloWorld;
begin
  writeln('Hello, World!');
end.
uses

Description: The uses keyword is used to include external libraries (units) that provide additional functionality for the program.

program HelloWorld;
uses crt;
begin
  clrscr;
  writeln('Hello, World!');
  readln;
end.
begin and end

Description: These keywords define a block of code. The begin marks the start and the end marks the end.

program HelloWorld;
begin
  writeln('This is the start of a program.');
  writeln('This is the second line of the program.');
end.
var

Description: The var keyword is used to declare variables that will store data that may change during the program's execution.

program VariableExample;
var
  num1, num2, sum: Integer;
begin
  num1 := 5;
  num2 := 10;
  sum := num1 + num2;
  writeln('The sum is: ', sum);
end.
const

Description: The const keyword defines constant values that cannot be changed during the program's execution.

program ConstantExample;
const
  Pi = 3.14159;
begin
  writeln('The value of Pi is: ', Pi);
end.
if and then

Description: The if statement is used to execute a block of code conditionally, and the then specifies the action if the condition is true.

program IfExample;
var
  age: Integer;
begin
  age := 20;
  if age >= 18 then
    writeln('You are an adult.');
end.
else

Description: The else keyword is used to specify an alternative block of code to execute if the condition is false.

program IfElseExample;
var
  age: Integer;
begin
  age := 16;
  if age >= 18 then
    writeln('You are an adult.')
  else
    writeln('You are a minor.');
end.
for and to

Description: The for loop is used to iterate over a range of values, and the to keyword defines the upper limit.

program ForLoopExample;
var
  i: Integer;
begin
  for i := 1 to 5 do
    writeln('Iteration ', i);
end.
while and do

Description: The while loop runs a block of code as long as the condition is true.

program WhileLoopExample;
var
  i: Integer;
begin
  i := 1;
  while i <= 5 do
  begin
    writeln('Iteration ', i);
    i := i + 1;
  end;
end.
repeat and until

Description: The repeat loop executes a block of code at least once and continues as long as the condition specified by until is false.

program RepeatUntilExample;
var
  i: Integer;
begin
  i := 1;
  repeat
    writeln('Iteration ', i);
    i := i + 1;
  until i > 5;
end.
function

Description: A function performs a task and returns a value.

program FunctionExample;
function AddNumbers(a, b: Integer): Integer;
begin
  AddNumbers := a + b;
end;

begin
  writeln('Sum of 5 and 7 is: ', AddNumbers(5, 7));
end.
procedure

Description: A procedure performs a task but does not return a value.

program ProcedureExample;
procedure Greet(name: String);
begin
  writeln('Hello, ', name);
end;

begin
  Greet('Alice');
  Greet('Bob');
end.
exit

Description: The exit keyword is used to terminate a program or a procedure.

program ExitExample;
begin
  writeln('Before exit');
  exit;
  writeln('This will not be printed.');
end.
type

Description: The type keyword defines new types, like records and arrays.

program TypeExample;
type
  Point = record
    x, y: Integer;
  end;
var
  p: Point;
begin
  p.x := 5;
  p.y := 10;
  writeln('Point coordinates: ', p.x, ', ', p.y);
end.
array

Description: An array is a collection of variables of the same type.

program ArrayExample;
var
  nums: array[1..5] of Integer;
  i: Integer;
begin
  nums[1] := 10;
  nums[2] := 20;
  nums[3] := 30;
  nums[4] := 40;
  nums[5] := 50;
  
  for i := 1 to 5 do
    writeln('nums[', i, ']: ', nums[i]);
end.
with

Description: The with keyword simplifies access to fields of a record or object.

program WithExample;
type
  Point = record
    x, y: Integer;
  end;
var
  p: Point;
begin
  p.x := 5;
  p.y := 10;
  with p do
  begin
    writeln('x: ', x, ', y: ', y);
  end;
end.
case

Description: The case keyword is used to execute a block of code based on the value of a variable.

program CaseExample;
var
  grade: Char;
begin
  grade := 'A';
  case grade of
    'A': writeln('Excellent!');
    'B': writeln('Good!');
    'C': writeln('Average!');
    'D': writeln('Poor!');
    'F': writeln('Fail!');
  else
    writeln('Invalid grade');
  end;
end.
